# Example values file for Frigate deployment
# Copy this file to values.yaml and customize for your environment

# Production values example
---
environment: production
app:
  namespace: frigate-prod
  replicas: 1
  image:
    tag: 0.12.0-stable

service:
  type: LoadBalancer
  loadBalancerIP: 10.10.10.100

ingress:
  hostname: frigate.production.example.com
  clusterIssuer: letsencrypt-prod

storage:
  size: 500Gi
  storageClass: fast-ssd

mqtt:
  host: mqtt.production.example.com
  user: frigate-prod

---
# Development values example
environment: development
app:

  replicas: 1
  image:
    tag: 0.12.0-beta

  enabled: false

  size: 20Gi
  storageClass: standard

  host: mqtt.dev.local
  user: frigate-dev

  timezone: UTC

---
